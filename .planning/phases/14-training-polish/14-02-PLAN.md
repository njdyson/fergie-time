**Create `src/season/coachingReport.ts`:**

```typescript
import type { PlayerState, PlayerAttributes } from '../simulation/types.ts';
import type { TrainingDeltas, TrainingDayPlan } from './season.ts';
import { DRILL_LABELS, DRILL_ATTRIBUTE_MAP } from './training.ts';
import type { DrillType } from './training.ts';

interface CoachingReportParams {
  subject: string;
  body: string;
  from: string;
  category: 'general';
}

/**
 * Generate a coaching report email after a training day.
 * Returns null if the day was a rest day or no training occurred.
 *
 * @param dayPlan - The drill or 'rest' for the day that just completed
 * @param dayNumber - 1-based day number (currentDay before increment, +1 for display)
 * @param squad - The player team's squad AFTER training was applied
 * @param deltas - The cumulative training deltas (we extract single-day deltas from the drill's targeted attributes)
 * @param squadBefore - The player team's squad BEFORE training was applied (to compute single-day gains)
 */
export function generateCoachingReport(
  dayPlan: TrainingDayPlan,
  dayNumber: number,
  squad: PlayerState[],
  squadBefore: PlayerState[],
): CoachingReportParams | null
```

The function:
1. If `dayPlan === 'rest'`, return null.
2. Get the drill label from `DRILL_LABELS[dayPlan]`.
3. Get targeted attributes from `DRILL_ATTRIBUTE_MAP[dayPlan]`.
4. Compute per-player gains by comparing `squadBefore[i].attributes[attr]` vs `squad[i].attributes[attr]` for each targeted attribute.
5. Sort players by total gain descending. Take top 3 as "standout improvers".
6. Build subject: `"Training Report: ${drillLabel} — Day ${dayNumber}"`
7. Build body (plain text, will be rendered as HTML in inbox):
   ```
   Today's session: ${drillLabel} drill

   Squad participation: ${squad.length} players

   Standout improvers:
   • ${name} — ${attrList} (+${totalGainPct}%)
   • ${name} — ${attrList} (+${totalGainPct}%)
   • ${name} — ${attrList} (+${totalGainPct}%)
   ```
   Where attrList shows the attribute names that improved (e.g., "Pace, Stamina, Strength") and totalGainPct is the sum of gains rounded to 1 decimal as percentage points.
8. Return `{ subject, body, from: 'Coaching Staff', category: 'general' }`.

Use `camelToTitle` or a simple mapping for attribute display names — import or inline a helper that converts 'pace' → 'Pace', 'stamina' → 'Stamina', etc.

   ```typescript
   hubScreenView.onContinue(() => {
     // Capture pre-advance state for coaching report
     const prevDay = seasonState.currentDay;
     const dayPlan = seasonState.trainingSchedule?.[prevDay] ?? 'rest';
     const playerTeam = seasonState.teams.find(t => t.isPlayerTeam)!;
     const squadBefore = playerTeam.squad;

     const result = advanceDay(seasonState);
     seasonState = result.state;

     // COACH-01: Generate coaching report email after training
     if (dayPlan !== 'rest') {
       const playerTeamAfter = seasonState.teams.find(t => t.isPlayerTeam)!;
       const report = generateCoachingReport(dayPlan, prevDay + 1, playerTeamAfter.squad, squadBefore);
       if (report) {
         seasonState = {
           ...seasonState,
           inbox: sendMessage(seasonState.inbox, {
             ...report,
             matchday: seasonState.currentMatchday,
           }),
         };
       }
     }

     // Auto-save after each day advance
     saveGame(serializeState(seasonState), 1).catch(err => console.error('Auto-save (day advance) failed:', err));
     updateCurrentScreen();
   });
   ```
